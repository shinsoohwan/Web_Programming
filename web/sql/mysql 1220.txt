drop database web;
create database web;
use web;

create table member
(
	id varchar(20) primary key,
	password varchar(40),
	mtype varchar(10)
);

create table seller
(
	s_id varchar(20),
	name varchar(20),
	phone varchar(20),
	bank varchar(15),
	account varchar(30)
);

create table buyer
(
	b_id varchar(20),
	name varchar(20),
	phone varchar(20),
	address varchar(100)
);

create table product
(
	p_id int primary key auto_increment,
	s_id varchar(20),
	name varchar(20),
	category1 varchar(20) not null,
	category2 varchar(20),
	price int,
	measure varchar(20),
	start_time date,
	content text,
	img_dir varchar(40)
);

create table orders
(
	o_id int primary key auto_increment,
	p_id int,
	s_id varchar(20),
	b_id varchar(20),
	numbers int,
	price int,
	order_time date,
	status varchar(20),
	delivery varchar(30)
);

create table notice
(
	numbers int primary key auto_increment,
	write_date date,
	title varchar(100),
	content text,
	img_dir varchar(40)
);
